# הברכות

> דברים כח א: וְהָיָה אִם שָׁמוֹעַ תִּשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ לִשְׁמֹר לַעֲשׂוֹת אֶת כָּל מִצְוֹתָיו אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם וּנְתָנְךָ יְהוָה אֱלֹהֶיךָ עֶלְיוֹן עַל כָּל גּוֹיֵי הָאָרֶץ.  
> דברים כח ב: וּבָאוּ עָלֶיךָ כָּל הַבְּרָכוֹת הָאֵלֶּה וְהִשִּׂיגֻךָ:  כִּי תִשְׁמַע בְּקוֹל יְהוָה אֱלֹהֶיךָ.  
> דברים כח ג: בָּרוּךְ אַתָּה בָּעִיר; וּבָרוּךְ אַתָּה בַּשָּׂדֶה.  
> דברים כח ד: בָּרוּךְ פְּרִי בִטְנְךָ וּפְרִי אַדְמָתְךָ וּפְרִי בְהֶמְתֶּךָ שְׁגַר אֲלָפֶיךָ וְעַשְׁתְּרוֹת צֹאנֶךָ.  
> דברים כח ה: בָּרוּךְ טַנְאֲךָ וּמִשְׁאַרְתֶּךָ.  
> דברים כח ו: בָּרוּךְ אַתָּה בְּבֹאֶךָ; וּבָרוּךְ אַתָּה בְּצֵאתֶךָ.  
> דברים כח ז: יִתֵּן יְהוָה אֶת אֹיְבֶיךָ הַקָּמִים עָלֶיךָ נִגָּפִים לְפָנֶיךָ:  בְּדֶרֶךְ אֶחָד יֵצְאוּ אֵלֶיךָ וּבְשִׁבְעָה דְרָכִים יָנוּסוּ לְפָנֶיךָ.  
> דברים כח ח: יְצַו יְהוָה אִתְּךָ אֶת הַבְּרָכָה בַּאֲסָמֶיךָ וּבְכֹל מִשְׁלַח יָדֶךָ; וּבֵרַכְךָ בָּאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ.  
> דברים כח ט: יְקִימְךָ יְהוָה לוֹ לְעַם קָדוֹשׁ כַּאֲשֶׁר נִשְׁבַּע לָךְ:  כִּי תִשְׁמֹר אֶת מִצְוֹת יְהוָה אֱלֹהֶיךָ וְהָלַכְתָּ בִּדְרָכָיו.  
> דברים כח י: וְרָאוּ כָּל עַמֵּי הָאָרֶץ כִּי שֵׁם יְהוָה נִקְרָא עָלֶיךָ; וְיָרְאוּ מִמֶּךָּ.  
> דברים כח יא: וְהוֹתִרְךָ יְהוָה לְטוֹבָה בִּפְרִי בִטְנְךָ וּבִפְרִי בְהֶמְתְּךָ וּבִפְרִי אַדְמָתֶךָ עַל הָאֲדָמָה אֲשֶׁר נִשְׁבַּע יְהוָה לַאֲבֹתֶיךָ לָתֶת לָךְ.  
> דברים כח יב: יִפְתַּח יְהוָה לְךָ אֶת אוֹצָרוֹ הַטּוֹב אֶת הַשָּׁמַיִם לָתֵת מְטַר אַרְצְךָ בְּעִתּוֹ וּלְבָרֵךְ אֵת כָּל מַעֲשֵׂה יָדֶךָ; וְהִלְוִיתָ גּוֹיִם רַבִּים וְאַתָּה לֹא תִלְוֶה.  
> דברים כח יג: וּנְתָנְךָ יְהוָה לְרֹאשׁ וְלֹא לְזָנָב וְהָיִיתָ רַק לְמַעְלָה וְלֹא תִהְיֶה לְמָטָּה:  כִּי תִשְׁמַע אֶל מִצְוֹת יְהוָה אֱלֹהֶיךָ אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לִשְׁמֹר וְלַעֲשׂוֹת.  
> דברים כח יד: וְלֹא תָסוּר מִכָּל הַדְּבָרִים אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם הַיּוֹם יָמִין וּשְׂמֹאול:  לָלֶכֶת אַחֲרֵי אֱלֹהִים אֲחֵרִים לְעָבְדָם.
