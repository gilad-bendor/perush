# רשימת דינים

> דברים כב ה: לֹא יִהְיֶה כְלִי גֶבֶר עַל אִשָּׁה וְלֹא יִלְבַּשׁ גֶּבֶר שִׂמְלַת אִשָּׁה:  כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ כָּל עֹשֵׂה אֵלֶּה.  
> דברים כב ו: כִּי יִקָּרֵא קַן צִפּוֹר לְפָנֶיךָ בַּדֶּרֶךְ בְּכָל עֵץ אוֹ עַל הָאָרֶץ אֶפְרֹחִים אוֹ בֵיצִים וְהָאֵם רֹבֶצֶת עַל הָאֶפְרֹחִים אוֹ עַל הַבֵּיצִים לֹא תִקַּח הָאֵם עַל הַבָּנִים.  
> דברים כב ז: שַׁלֵּחַ תְּשַׁלַּח אֶת הָאֵם וְאֶת הַבָּנִים תִּקַּח לָךְ לְמַעַן יִיטַב לָךְ וְהַאֲרַכְתָּ יָמִים.  
> דברים כב ח: כִּי תִבְנֶה בַּיִת חָדָשׁ וְעָשִׂיתָ מַעֲקֶה לְגַגֶּךָ; וְלֹא תָשִׂים דָּמִים בְּבֵיתֶךָ כִּי יִפֹּל הַנֹּפֵל מִמֶּנּוּ.  
> דברים כב ט: לֹא תִזְרַע כַּרְמְךָ כִּלְאָיִם:  פֶּן תִּקְדַּשׁ הַמְלֵאָה הַזֶּרַע אֲשֶׁר תִּזְרָע וּתְבוּאַת הַכָּרֶם.  
> דברים כב י: לֹא תַחֲרֹשׁ בְּשׁוֹר וּבַחֲמֹר יַחְדָּו.  
> דברים כב יא: לֹא תִלְבַּשׁ שַׁעַטְנֵז צֶמֶר וּפִשְׁתִּים יַחְדָּו.  
> דברים כב יב: גְּדִלִים תַּעֲשֶׂה לָּךְ עַל אַרְבַּע כַּנְפוֹת כְּסוּתְךָ אֲשֶׁר תְּכַסֶּה בָּהּ.  

# עלילת נערה בתולה
> דברים כב יג: כִּי יִקַּח אִישׁ אִשָּׁה; וּבָא אֵלֶיהָ וּשְׂנֵאָהּ.  
> דברים כב יד: וְשָׂם לָהּ עֲלִילֹת דְּבָרִים וְהוֹצִא עָלֶיהָ שֵׁם רָע; וְאָמַר אֶת הָאִשָּׁה הַזֹּאת לָקַחְתִּי וָאֶקְרַב אֵלֶיהָ וְלֹא מָצָאתִי לָהּ בְּתוּלִים.  
> דברים כב טו: וְלָקַח אֲבִי הַנַּעֲרָ וְאִמָּהּ; וְהוֹצִיאוּ אֶת בְּתוּלֵי הַנַּעֲרָ אֶל זִקְנֵי הָעִיר הַשָּׁעְרָה.  
> דברים כב טז: וְאָמַר אֲבִי הַנַּעֲרָ אֶל הַזְּקֵנִים:  אֶת בִּתִּי נָתַתִּי לָאִישׁ הַזֶּה לְאִשָּׁה וַיִּשְׂנָאֶהָ.  
> דברים כב יז: וְהִנֵּה הוּא שָׂם עֲלִילֹת דְּבָרִים לֵאמֹר לֹא מָצָאתִי לְבִתְּךָ בְּתוּלִים וְאֵלֶּה בְּתוּלֵי בִתִּי; וּפָרְשׂוּ הַשִּׂמְלָה לִפְנֵי זִקְנֵי הָעִיר.  
> דברים כב יח: וְלָקְחוּ זִקְנֵי הָעִיר הַהִוא אֶת הָאִישׁ; וְיִסְּרוּ אֹתוֹ.  
> דברים כב יט: וְעָנְשׁוּ אֹתוֹ מֵאָה כֶסֶף וְנָתְנוּ לַאֲבִי הַנַּעֲרָה כִּי הוֹצִיא שֵׁם רָע עַל בְּתוּלַת יִשְׂרָאֵל; וְלוֹ תִהְיֶה לְאִשָּׁה לֹא יוּכַל לְשַׁלְּחָהּ כָּל יָמָיו.  
> דברים כב כ: וְאִם אֱמֶת הָיָה הַדָּבָר הַזֶּה:  לֹא נִמְצְאוּ בְתוּלִים לַנַּעֲרָ.  
> דברים כב כא: וְהוֹצִיאוּ אֶת הַנַּעֲרָ אֶל פֶּתַח בֵּית אָבִיהָ וּסְקָלוּהָ אַנְשֵׁי עִירָהּ בָּאֲבָנִים וָמֵתָה כִּי עָשְׂתָה נְבָלָה בְּיִשְׂרָאֵל לִזְנוֹת בֵּית אָבִיהָ; וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ.  

# דיני ניאוף ואונס
> דברים כב כב: כִּי יִמָּצֵא אִישׁ שֹׁכֵב עִם אִשָּׁה בְעֻלַת בַּעַל וּמֵתוּ גַּם שְׁנֵיהֶם הָאִישׁ הַשֹּׁכֵב עִם הָאִשָּׁה וְהָאִשָּׁה; וּבִעַרְתָּ הָרָע מִיִּשְׂרָאֵל.  
> דברים כב כג: כִּי יִהְיֶה נַעֲרָ בְתוּלָה מְאֹרָשָׂה לְאִישׁ; וּמְצָאָהּ אִישׁ בָּעִיר וְשָׁכַב עִמָּהּ.  
> דברים כב כד: וְהוֹצֵאתֶם אֶת שְׁנֵיהֶם אֶל שַׁעַר הָעִיר הַהִוא וּסְקַלְתֶּם אֹתָם בָּאֲבָנִים וָמֵתוּ אֶת הַנַּעֲרָ עַל דְּבַר אֲשֶׁר לֹא צָעֲקָה בָעִיר וְאֶת הָאִישׁ עַל דְּבַר אֲשֶׁר עִנָּה אֶת אֵשֶׁת רֵעֵהוּ; וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ.  
> דברים כב כה: וְאִם בַּשָּׂדֶה יִמְצָא הָאִישׁ אֶת הַנַּעֲרָ הַמְאֹרָשָׂה וְהֶחֱזִיק בָּהּ הָאִישׁ וְשָׁכַב עִמָּהּ:  וּמֵת הָאִישׁ אֲשֶׁר שָׁכַב עִמָּהּ לְבַדּוֹ.  
> דברים כב כו: וְלַנַּעֲרָ לֹא תַעֲשֶׂה דָבָר אֵין לַנַּעֲרָ חֵטְא מָוֶת:  כִּי כַּאֲשֶׁר יָקוּם אִישׁ עַל רֵעֵהוּ וּרְצָחוֹ נֶפֶשׁ כֵּן הַדָּבָר הַזֶּה.  
> דברים כב כז: כִּי בַשָּׂדֶה מְצָאָהּ; צָעֲקָה הַנַּעֲרָ הַמְאֹרָשָׂה וְאֵין מוֹשִׁיעַ לָהּ.  
> דברים כב כח: כִּי יִמְצָא אִישׁ נַעֲרָ בְתוּלָה אֲשֶׁר לֹא אֹרָשָׂה וּתְפָשָׂהּ וְשָׁכַב עִמָּהּ; וְנִמְצָאוּ.  
> דברים כב כט: וְנָתַן הָאִישׁ הַשֹּׁכֵב עִמָּהּ לַאֲבִי הַנַּעֲרָ חֲמִשִּׁים כָּסֶף; וְלוֹ תִהְיֶה לְאִשָּׁה תַּחַת אֲשֶׁר עִנָּהּ לֹא יוּכַל שַׁלְּחָהּ כָּל יָמָיו.  

# רשימת דינים
> דברים כג א: לֹא יִקַּח אִישׁ אֶת אֵשֶׁת אָבִיו; וְלֹא יְגַלֶּה כְּנַף אָבִיו.  
> דברים כג ב: לֹא יָבֹא פְצוּעַ דַּכָּא וּכְרוּת שָׁפְכָה בִּקְהַל יְהוָה.  

# היחס לממזר, עמוני ומואבי, אדומי
> דברים כג ג: לֹא יָבֹא מַמְזֵר בִּקְהַל יְהוָה:  גַּם דּוֹר עֲשִׂירִי לֹא יָבֹא לוֹ בִּקְהַל יְהוָה.  
> דברים כג ד: לֹא יָבֹא עַמּוֹנִי וּמוֹאָבִי בִּקְהַל יְהוָה:  גַּם דּוֹר עֲשִׂירִי לֹא יָבֹא לָהֶם בִּקְהַל יְהוָה עַד עוֹלָם.  
> דברים כג ה: עַל דְּבַר אֲשֶׁר לֹא קִדְּמוּ אֶתְכֶם בַּלֶּחֶם וּבַמַּיִם בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם; וַאֲשֶׁר שָׂכַר עָלֶיךָ אֶת בִּלְעָם בֶּן בְּעוֹר מִפְּתוֹר אֲרַם נַהֲרַיִם לְקַלְלֶךָּ.  
> דברים כג ו: וְלֹא אָבָה יְהוָה אֱלֹהֶיךָ לִשְׁמֹעַ אֶל בִּלְעָם וַיַּהֲפֹךְ יְהוָה אֱלֹהֶיךָ לְּךָ אֶת הַקְּלָלָה לִבְרָכָה:  כִּי אֲהֵבְךָ יְהוָה אֱלֹהֶיךָ.  
> דברים כג ז: לֹא תִדְרֹשׁ שְׁלֹמָם וְטֹבָתָם כָּל יָמֶיךָ לְעוֹלָם.  
> דברים כג ח: לֹא תְתַעֵב אֲדֹמִי כִּי אָחִיךָ הוּא; לֹא תְתַעֵב מִצְרִי כִּי גֵר הָיִיתָ בְאַרְצוֹ.  
> דברים כג ט: בָּנִים אֲשֶׁר יִוָּלְדוּ לָהֶם דּוֹר שְׁלִישִׁי יָבֹא לָהֶם בִּקְהַל יְהוָה.  

# קדושת המחנה
> דברים כג י: כִּי תֵצֵא מַחֲנֶה עַל אֹיְבֶיךָ:  וְנִשְׁמַרְתָּ מִכֹּל דָּבָר רָע.  
> דברים כג יא: כִּי יִהְיֶה בְךָ אִישׁ אֲשֶׁר לֹא יִהְיֶה טָהוֹר מִקְּרֵה לָיְלָה וְיָצָא אֶל מִחוּץ לַמַּחֲנֶה לֹא יָבֹא אֶל תּוֹךְ הַמַּחֲנֶה.  
> דברים כג יב: וְהָיָה לִפְנוֹת עֶרֶב יִרְחַץ בַּמָּיִם; וּכְבֹא הַשֶּׁמֶשׁ יָבֹא אֶל תּוֹךְ הַמַּחֲנֶה.  
> דברים כג יג: וְיָד תִּהְיֶה לְךָ מִחוּץ לַמַּחֲנֶה; וְיָצָאתָ שָּׁמָּה חוּץ.  
> דברים כג יד: וְיָתֵד תִּהְיֶה לְךָ עַל אֲזֵנֶךָ; וְהָיָה בְּשִׁבְתְּךָ חוּץ וְחָפַרְתָּה בָהּ וְשַׁבְתָּ וְכִסִּיתָ אֶת צֵאָתֶךָ.  
> דברים כג טו: כִּי יְהוָה אֱלֹהֶיךָ מִתְהַלֵּךְ בְּקֶרֶב מַחֲנֶךָ לְהַצִּילְךָ וְלָתֵת אֹיְבֶיךָ לְפָנֶיךָ וְהָיָה מַחֲנֶיךָ קָדוֹשׁ:  וְלֹא יִרְאֶה בְךָ עֶרְוַת דָּבָר וְשָׁב מֵאַחֲרֶיךָ.  

# דינים של סולידריות חברתית
> דברים כג טז: לֹא תַסְגִּיר עֶבֶד אֶל אֲדֹנָיו אֲשֶׁר יִנָּצֵל אֵלֶיךָ מֵעִם אֲדֹנָיו.  
> דברים כג יז: עִמְּךָ יֵשֵׁב בְּקִרְבְּךָ בַּמָּקוֹם אֲשֶׁר יִבְחַר בְּאַחַד שְׁעָרֶיךָ בַּטּוֹב לוֹ; לֹא תּוֹנֶנּוּ.  
> דברים כג יח: לֹא תִהְיֶה קְדֵשָׁה מִבְּנוֹת יִשְׂרָאֵל; וְלֹא יִהְיֶה קָדֵשׁ מִבְּנֵי יִשְׂרָאֵל.  
> דברים כג יט: לֹא תָבִיא אֶתְנַן זוֹנָה וּמְחִיר כֶּלֶב בֵּית יְהוָה אֱלֹהֶיךָ לְכָל נֶדֶר:  כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ גַּם שְׁנֵיהֶם.  
> דברים כג כ: לֹא תַשִּׁיךְ לְאָחִיךָ נֶשֶׁךְ כֶּסֶף נֶשֶׁךְ אֹכֶל:  נֶשֶׁךְ כָּל דָּבָר אֲשֶׁר יִשָּׁךְ.  
> דברים כג כא: לַנָּכְרִי תַשִּׁיךְ וּלְאָחִיךָ לֹא תַשִּׁיךְ לְמַעַן יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל מִשְׁלַח יָדֶךָ עַל הָאָרֶץ אֲשֶׁר אַתָּה בָא שָׁמָּה לְרִשְׁתָּהּ.  

# נדרים והמלצה להמנע
> דברים כג כב: כִּי תִדֹּר נֶדֶר לַיהוָה אֱלֹהֶיךָ לֹא תְאַחֵר לְשַׁלְּמוֹ:  כִּי דָּרֹשׁ יִדְרְשֶׁנּוּ יְהוָה אֱלֹהֶיךָ מֵעִמָּךְ וְהָיָה בְךָ חֵטְא.  
> דברים כג כג: וְכִי תֶחְדַּל לִנְדֹּר לֹא יִהְיֶה בְךָ חֵטְא.  
> דברים כג כד: מוֹצָא שְׂפָתֶיךָ תִּשְׁמֹר וְעָשִׂיתָ:  כַּאֲשֶׁר נָדַרְתָּ לַיהוָה אֱלֹהֶיךָ נְדָבָה אֲשֶׁר דִּבַּרְתָּ בְּפִיךָ.  

# אישור אכילת נפש מרכוש של אחר
> דברים כג כה: כִּי תָבֹא בְּכֶרֶם רֵעֶךָ וְאָכַלְתָּ עֲנָבִים כְּנַפְשְׁךָ שָׂבְעֶךָ; וְאֶל כֶּלְיְךָ לֹא תִתֵּן.  
> דברים כג כו: כִּי תָבֹא בְּקָמַת רֵעֶךָ וְקָטַפְתָּ מְלִילֹת בְּיָדֶךָ; וְחֶרְמֵשׁ לֹא תָנִיף עַל קָמַת רֵעֶךָ.  

# איסור נישואין לאחר גירושין
> דברים כד א: כִּי יִקַּח אִישׁ אִשָּׁה וּבְעָלָהּ; וְהָיָה אִם לֹא תִמְצָא חֵן בְּעֵינָיו כִּי מָצָא בָהּ עֶרְוַת דָּבָר וְכָתַב לָהּ סֵפֶר כְּרִיתֻת וְנָתַן בְּיָדָהּ וְשִׁלְּחָהּ מִבֵּיתוֹ.  
> דברים כד ב: וְיָצְאָה מִבֵּיתוֹ; וְהָלְכָה וְהָיְתָה לְאִישׁ אַחֵר.  
> דברים כד ג: וּשְׂנֵאָהּ הָאִישׁ הָאַחֲרוֹן וְכָתַב לָהּ סֵפֶר כְּרִיתֻת וְנָתַן בְּיָדָהּ וְשִׁלְּחָהּ מִבֵּיתוֹ; אוֹ כִי יָמוּת הָאִישׁ הָאַחֲרוֹן אֲשֶׁר לְקָחָהּ לוֹ לְאִשָּׁה.  
> דברים כד ד: לֹא יוּכַל בַּעְלָהּ הָרִאשׁוֹן אֲשֶׁר שִׁלְּחָהּ לָשׁוּב לְקַחְתָּהּ לִהְיוֹת לוֹ לְאִשָּׁה אַחֲרֵי אֲשֶׁר הֻטַּמָּאָה כִּי תוֹעֵבָה הִוא לִפְנֵי יְהוָה; וְלֹא תַחֲטִיא אֶת הָאָרֶץ אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לְךָ נַחֲלָה.  

# רשימת דינים
> דברים כד ה: כִּי יִקַּח אִישׁ אִשָּׁה חֲדָשָׁה לֹא יֵצֵא בַּצָּבָא וְלֹא יַעֲבֹר עָלָיו לְכָל דָּבָר:  נָקִי יִהְיֶה לְבֵיתוֹ שָׁנָה אֶחָת וְשִׂמַּח אֶת אִשְׁתּוֹ אֲשֶׁר לָקָח.  
> דברים כד ו: לֹא יַחֲבֹל רֵחַיִם וָרָכֶב:  כִּי נֶפֶשׁ הוּא חֹבֵל.  
> דברים כד ז: כִּי יִמָּצֵא אִישׁ גֹּנֵב נֶפֶשׁ מֵאֶחָיו מִבְּנֵי יִשְׂרָאֵל וְהִתְעַמֶּר בּוֹ וּמְכָרוֹ וּמֵת הַגַּנָּב הַהוּא וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ.  
> דברים כד ח: הִשָּׁמֶר בְּנֶגַע הַצָּרַעַת לִשְׁמֹר מְאֹד וְלַעֲשׂוֹת:  כְּכֹל אֲשֶׁר יוֹרוּ אֶתְכֶם הַכֹּהֲנִים הַלְוִיִּם כַּאֲשֶׁר צִוִּיתִם תִּשְׁמְרוּ לַעֲשׂוֹת.  

# כבוד לחייב - איסור נשיית עבוט מתוך ניתוק רגשי
> דברים כד ט: זָכוֹר אֵת אֲשֶׁר עָשָׂה יְהוָה אֱלֹהֶיךָ לְמִרְיָם בַּדֶּרֶךְ בְּצֵאתְכֶם מִמִּצְרָיִם.  
> דברים כד י: כִּי תַשֶּׁה בְרֵעֲךָ מַשַּׁאת מְאוּמָה לֹא תָבֹא אֶל בֵּיתוֹ לַעֲבֹט עֲבֹטוֹ.  
> דברים כד יא: בַּחוּץ תַּעֲמֹד; וְהָאִישׁ אֲשֶׁר אַתָּה נֹשֶׁה בוֹ יוֹצִיא אֵלֶיךָ אֶת הַעֲבוֹט הַחוּצָה.  
> דברים כד יב: וְאִם אִישׁ עָנִי הוּא לֹא תִשְׁכַּב בַּעֲבֹטוֹ.  
> דברים כד יג: הָשֵׁב תָּשִׁיב לוֹ אֶת הַעֲבוֹט כְּבוֹא הַשֶּׁמֶשׁ וְשָׁכַב בְּשַׂלְמָתוֹ וּבֵרְכֶךָּ; וּלְךָ תִּהְיֶה צְדָקָה לִפְנֵי יְהוָה אֱלֹהֶיךָ.  

# דינים של סולידריות חברתית, חמלה וצדק
> דברים כד יד: לֹא תַעֲשֹׁק שָׂכִיר עָנִי וְאֶבְיוֹן מֵאַחֶיךָ אוֹ מִגֵּרְךָ אֲשֶׁר בְּאַרְצְךָ בִּשְׁעָרֶיךָ.  
> דברים כד טו: בְּיוֹמוֹ תִתֵּן שְׂכָרוֹ וְלֹא תָבוֹא עָלָיו הַשֶּׁמֶשׁ כִּי עָנִי הוּא וְאֵלָיו הוּא נֹשֵׂא אֶת נַפְשׁוֹ; וְלֹא יִקְרָא עָלֶיךָ אֶל יְהוָה וְהָיָה בְךָ חֵטְא.  
> דברים כד טז: לֹא יוּמְתוּ אָבוֹת עַל בָּנִים וּבָנִים לֹא יוּמְתוּ עַל אָבוֹת:  אִישׁ בְּחֶטְאוֹ יוּמָתוּ.  
> דברים כד יז: לֹא תַטֶּה מִשְׁפַּט גֵּר יָתוֹם; וְלֹא תַחֲבֹל בֶּגֶד אַלְמָנָה.  
> דברים כד יח: וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּמִצְרַיִם וַיִּפְדְּךָ יְהוָה אֱלֹהֶיךָ מִשָּׁם; עַל כֵּן אָנֹכִי מְצַוְּךָ לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה.  
> דברים כד יט: כִּי תִקְצֹר קְצִירְךָ בְשָׂדֶךָ וְשָׁכַחְתָּ עֹמֶר בַּשָּׂדֶה לֹא תָשׁוּב לְקַחְתּוֹ לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה:  לְמַעַן יְבָרֶכְךָ יְהוָה אֱלֹהֶיךָ בְּכֹל מַעֲשֵׂה יָדֶיךָ.  
> דברים כד כ: כִּי תַחְבֹּט זֵיתְךָ לֹא תְפָאֵר אַחֲרֶיךָ:  לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה.  
> דברים כד כא: כִּי תִבְצֹר כַּרְמְךָ לֹא תְעוֹלֵל אַחֲרֶיךָ:  לַגֵּר לַיָּתוֹם וְלָאַלְמָנָה יִהְיֶה.  
> דברים כד כב: וְזָכַרְתָּ כִּי עֶבֶד הָיִיתָ בְּאֶרֶץ מִצְרָיִם; עַל כֵּן אָנֹכִי מְצַוְּךָ לַעֲשׂוֹת אֶת הַדָּבָר הַזֶּה.  
> דברים כה א: כִּי יִהְיֶה רִיב בֵּין אֲנָשִׁים וְנִגְּשׁוּ אֶל הַמִּשְׁפָּט וּשְׁפָטוּם; וְהִצְדִּיקוּ אֶת הַצַּדִּיק וְהִרְשִׁיעוּ אֶת הָרָשָׁע.  
> דברים כה ב: וְהָיָה אִם בִּן הַכּוֹת הָרָשָׁע וְהִפִּילוֹ הַשֹּׁפֵט וְהִכָּהוּ לְפָנָיו כְּדֵי רִשְׁעָתוֹ בְּמִסְפָּר.  
> דברים כה ג: אַרְבָּעִים יַכֶּנּוּ לֹא יֹסִיף:  פֶּן יֹסִיף לְהַכֹּתוֹ עַל אֵלֶּה מַכָּה רַבָּה וְנִקְלָה אָחִיךָ לְעֵינֶיךָ.  
> דברים כה ד: לֹא תַחְסֹם שׁוֹר בְּדִישׁוֹ.  

# דיני ייבום
> דברים כה ה: כִּי יֵשְׁבוּ אַחִים יַחְדָּו וּמֵת אַחַד מֵהֶם וּבֵן אֵין לוֹ לֹא תִהְיֶה אֵשֶׁת הַמֵּת הַחוּצָה לְאִישׁ זָר:  יְבָמָהּ יָבֹא עָלֶיהָ וּלְקָחָהּ לוֹ לְאִשָּׁה וְיִבְּמָהּ.  
> דברים כה ו: וְהָיָה הַבְּכוֹר אֲשֶׁר תֵּלֵד יָקוּם עַל שֵׁם אָחִיו הַמֵּת; וְלֹא יִמָּחֶה שְׁמוֹ מִיִּשְׂרָאֵל.  
> דברים כה ז: וְאִם לֹא יַחְפֹּץ הָאִישׁ לָקַחַת אֶת יְבִמְתּוֹ; וְעָלְתָה יְבִמְתּוֹ הַשַּׁעְרָה אֶל הַזְּקֵנִים וְאָמְרָה מֵאֵן יְבָמִי לְהָקִים לְאָחִיו שֵׁם בְּיִשְׂרָאֵל לֹא אָבָה יַבְּמִי.  
> דברים כה ח: וְקָרְאוּ לוֹ זִקְנֵי עִירוֹ וְדִבְּרוּ אֵלָיו; וְעָמַד וְאָמַר לֹא חָפַצְתִּי לְקַחְתָּהּ.  
> דברים כה ט: וְנִגְּשָׁה יְבִמְתּוֹ אֵלָיו לְעֵינֵי הַזְּקֵנִים וְחָלְצָה נַעֲלוֹ מֵעַל רַגְלוֹ וְיָרְקָה בְּפָנָיו; וְעָנְתָה וְאָמְרָה כָּכָה יֵעָשֶׂה לָאִישׁ אֲשֶׁר לֹא יִבְנֶה אֶת בֵּית אָחִיו.  
> דברים כה י: וְנִקְרָא שְׁמוֹ בְּיִשְׂרָאֵל:  בֵּית חֲלוּץ הַנָּעַל.  

# רשימת דינים
> דברים כה יא: כִּי יִנָּצוּ אֲנָשִׁים יַחְדָּו אִישׁ וְאָחִיו וְקָרְבָה אֵשֶׁת הָאֶחָד לְהַצִּיל אֶת אִישָּׁהּ מִיַּד מַכֵּהוּ; וְשָׁלְחָה יָדָהּ וְהֶחֱזִיקָה בִּמְבֻשָׁיו.  
> דברים כה יב: וְקַצֹּתָה אֶת כַּפָּהּ:  לֹא תָחוֹס עֵינֶךָ.  
> דברים כה יג: לֹא יִהְיֶה לְךָ בְּכִיסְךָ אֶבֶן וָאָבֶן:  גְּדוֹלָה וּקְטַנָּה.  
> דברים כה יד: לֹא יִהְיֶה לְךָ בְּבֵיתְךָ אֵיפָה וְאֵיפָה:  גְּדוֹלָה וּקְטַנָּה.  
> דברים כה טו: אֶבֶן שְׁלֵמָה וָצֶדֶק יִהְיֶה לָּךְ אֵיפָה שְׁלֵמָה וָצֶדֶק יִהְיֶה לָּךְ לְמַעַן יַאֲרִיכוּ יָמֶיךָ עַל הָאֲדָמָה אֲשֶׁר יְהוָה אֱלֹהֶיךָ נֹתֵן לָךְ.  
> דברים כה טז: כִּי תוֹעֲבַת יְהוָה אֱלֹהֶיךָ כָּל עֹשֵׂה אֵלֶּה:  כֹּל עֹשֵׂה עָוֶל.
